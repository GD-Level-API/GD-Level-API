const { EmbedBuilder, AttachmentBuilder } = require('discord.js');

async execute(interaction) {
  const id    = interaction.options.getInteger('id');
  const data  = await fetch(`https://thumball.liamt.xyz/api/level?id=${id}`).then(r => r.json());

  if (data.error) return interaction.reply({ content: 'Nivel no encontrado', ephemeral: true });

  const card  = new AttachmentBuilder(data.urls.card, { name: 'card.png' });
  const embed = new EmbedBuilder()
    .setTitle(data.name)
    .setDescription(`por **${data.author}** · ${data.difficulty}`)
    .setImage('attachment://card.png')
    .setColor(0xf0c94e);

  await interaction.reply({ embeds: [embed], files: [card] });
}
import discord, aiohttp, io
from discord import app_commands

@tree.command(name="nivel")
async def nivel(interaction: discord.Interaction, id: int):
    async with aiohttp.ClientSession() as s:
        async with s.get(f"https://thumball.liamt.xyz/api/level?id={id}") as r:
            data = await r.json()
        async with s.get(data["urls"]["card"]) as r:
            png = await r.read()

    file  = discord.File(io.BytesIO(png), filename="card.png")
    embed = discord.Embed(
        title=data["name"],
        description=f"por **{data['author']}** · {data['difficulty']}",
        color=0xf0c94e
    ).set_image(url="attachment://card.png")
    await interaction.response.send_message(embed=embed, file=file)
// Discord.NET (C#) — SlashCommand handler
using Discord;
using Discord.WebSocket;
using System.Net.Http.Json;

private async Task HandleNivel(SocketSlashCommand cmd) {
    var id   = (long)cmd.Data.Options.First().Value;
    var http = new HttpClient();

    var data = await http.GetFromJsonAsync<JsonElement>(
        $"https://thumball.liamt.xyz/api/level?id={id}"
    );
    var png = await http.GetByteArrayAsync(
        data.GetProperty("urls").GetProperty("card").GetString()
    );

    var embed = new EmbedBuilder()
        .WithTitle(data.GetProperty("name").GetString())
        .WithImageUrl("attachment://card.png")
        .WithColor(new Color(0xf0c94e));

    await cmd.RespondWithFileAsync(
        new MemoryStream(png), "card.png",
        embed: embed.Build()
    );
}
// JDA (Java) — SlashCommandInteractionEvent
@Override
public void onSlashCommandInteraction(SlashCommandInteractionEvent event) {
    long id = event.getOption("id").getAsLong();

    var client = HttpClient.newHttpClient();

    // Fetch JSON
    var jsonBody = client
        .send(HttpRequest.newBuilder()
            .uri(URI.create("https://thumball.liamt.xyz/api/level?id=" + id))
            .build(),
            HttpResponse.BodyHandlers.ofString())
        .body();

    // Fetch card PNG
    byte[] png = client
        .send(HttpRequest.newBuilder()
            .uri(URI.create("https://thumball.liamt.xyz/api/card?id=" + id))
            .build(),
            HttpResponse.BodyHandlers.ofByteArray())
        .body();

    event.replyFiles(FileUpload.fromData(png, "card.png"))
        .addEmbeds(EmbedBuilder.fromData(jsonBody))
        .queue();
}

const Eris = require("eris");
const bot  = new Eris("TOKEN");

bot.on("interactionCreate", async (interaction) => {
  if (interaction.data.name !== "nivel") return;

  const id   = interaction.data.options[0].value;
  const data = await fetch(`https://thumball.liamt.xyz/api/level?id=${id}`)
                       .then(r => r.json());
  const png  = await fetch(data.urls.card)
                       .then(r => r.arrayBuffer());

  await interaction.createMessage({
    embeds: [{ title: data.name, image: { url: "attachment://card.png" } }],
    files:  [{ name: "card.png", file: Buffer.from(png) }],
  });
});
